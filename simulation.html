sim_mean_sd = function(n, mu = 3, sigma = 4) {
  
  sim_data = tibble(
    x = rnorm(n, mean = mu, sd = sigma) #generate data from normal distribution
  )
  
  sim_data %>% 
    summarize(
      mean = mean(x),
      sigma = sd(x)
    )
}
